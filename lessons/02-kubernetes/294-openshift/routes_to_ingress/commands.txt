######## API GATEWAY ########

cat > ingress-api-gateway.yml << EOF
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: ingress-api-gateway
  namespace: qsi-devops
spec:
  ingressClassName: nginx
  tls:
  - hosts:
    - devops-dev-api.qualifacts.org
    secretName: qualifacts-org-2023
  rules:
  - host: devops-dev-api.qualifacts.org
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: api-gateway
            port:
              number: 8083
EOF
oc apply -f ingress-api-gateway.yml
